# get_soup 함수 만들기
def get_soup(url):
    driver = webdriver.Chrome('./chromedriver')
    driver.implicitly_wait(3)
    driver.get(url)
    html = driver.page_source
    soup = BeautifulSoup(html, 'html.parser')
    driver.close()
    return soup

# 시청률 정보 담을 데이터프레임 생성
running = pd.DataFrame(columns = column_name)

# 시청률 정보, 일별로 데이터프레임 만들어서 running 데이터프레임에 append
for row in table[0].findAll("tr"):
    each_row = []
    for item in row.findAll("div"):
        each_row.append(item.text)
    tem = pd.DataFrame([each_row], columns=column_name)
    running = running.append(tem)

# 시청률 표시에 오타로 콤마가 들어간 경우 .으로 변경
running['닐슨'] = running['닐슨'].apply(lambda x : x.replace(",", "."))
running['TNMS'] = running['TNMS'].apply(lambda x : x.replace(",", "."))
running = running.iloc[1:, :]
